// ── SAVINGS CALCULATOR ────────────────────────
const calcRates = {
  'USD-NGN': 1567,
  'USD-KES': 129.4,
  'USD-GHS': 14.8,
  'GBP-NGN': 2040,
  'GBP-KES': 168.5,
  'GBP-GHS': 19.2,
  'CAD-NGN': 1143,
  'CAD-KES': 94.5,
  'CAD-GHS': 10.8,
  'AED-NGN': 426.8,
  'AED-KES': 35.2,
  'AED-GHS': 4.0,
};

const currSymbols = { NGN:'₦', KES:'KSh', GHS:'GH₵' };
const bankFeeRate = 0.05; // 5% bank margin
const walletFeeRate = 0.03; // 3% wallet fee

let calcAnimFrame;

function animateNumber(el, target, prefix, suffix) {
  const start = parseFloat(el.dataset.current||'0')||0;
  const duration = 600;
  const startTime = performance.now();
  cancelAnimationFrame(calcAnimFrame);
  function tick(now) {
    const t = Math.min((now - startTime)/duration, 1);
    const eased = 1 - Math.pow(1-t, 3);
    const current = start + (target-start)*eased;
    el.textContent = prefix + Math.round(current).toLocaleString() + suffix;
    el.dataset.current = current;
    if (t<1) calcAnimFrame = requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

function calcSavings() {
  const amount = parseFloat(document.getElementById('calc-amount').value)||0;
  const from = document.getElementById('calc-from').value;
  const to = document.getElementById('calc-to').value;
  const key = `${from}-${to}`;
  const rate = calcRates[key]||1567;
  const sym = currSymbols[to]||'';

  const xspRecv = Math.round(amount * rate);
  const bankRecv = Math.round(xspRecv * (1 - bankFeeRate));
  const walletRecv = Math.round(xspRecv * (1 - walletFeeRate));
  const savingsVsBank = xspRecv - bankRecv;
  const savingsVsWallet = xspRecv - walletRecv;

  animateNumber(document.getElementById('calc-savings-display'), savingsVsBank, sym, '');
  document.getElementById('calc-savings-sub').textContent = `on a ${from} ${amount.toLocaleString()} exchange`;
  document.getElementById('ct-xsp').textContent = sym + xspRecv.toLocaleString();
  document.getElementById('ct-bank').textContent = sym + bankRecv.toLocaleString();
  document.getElementById('ct-wallet').textContent = sym + walletRecv.toLocaleString();
  document.getElementById('ct-bank-diff').textContent = `−${sym}${savingsVsBank.toLocaleString()}`;
  document.getElementById('ct-wallet-diff').textContent = `−${sym}${savingsVsWallet.toLocaleString()}`;
}

const countObserver = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.isIntersecting && !entry.target.dataset.animated) {
      entry.target.dataset.animated = '1';
      const target = parseInt(entry.target.dataset.count || '0');
      if (!target) return;
      const duration = 1600;
      const startTime = performance.now();
      const el = entry.target;
      function tick(now) {
        const t = Math.min((now-startTime)/duration,1);
        const eased = 1-Math.pow(1-t,3);
        const cur = Math.round(target*eased);
        if (el.id==='stat-exchanges') el.textContent = cur.toLocaleString()+'+';
        if (el.id==='stat-volume') el.textContent = '$'+Math.round(cur/1e6)+'M+';
        if (t<1) requestAnimationFrame(tick);
      }
      requestAnimationFrame(tick);
    }
  });
}, { threshold: 0.5 });

document.querySelectorAll('[data-count]').forEach(el => countObserver.observe(el));
